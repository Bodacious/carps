   def connect
      until false 
         begin
            puts "Making IMAP connection for " + @username
            @imap = Net::IMAP.new(@server, 993, true)
            @imap.login(@username, @password)
            return
         rescue
            log "Could not connect to IMAP server", "Attempting to reconnect."
         end 
      end
   end
