   def read 
      # A reader
      reader = lambda do
         # Block 'till we get one
         while true
            @imap.select("inbox")
            messages = @imap.search(["UNSEEN"])
            if messages.empty?
               sleep 60
            else
               msg = messages[0]
               mails = @imap.fetch msg, "BODY[TEXT]"
               msg = mails[0].attr["BODY[TEXT]"]
               return msg.gsub "\r\n", ""
            end
         end
      end
      until false
