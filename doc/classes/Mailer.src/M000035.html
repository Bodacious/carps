   def handshake addr
      # Create a new peer
      peer = @peers[addr] = (Peer.new addr)
      # Send our key to the peer
      send addr, (Handshake.new @addr, @public_key)
      # Get the peer's key
      their_key = read :handshake, addr
      peer.your_key their_key.key
   end
