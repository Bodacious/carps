      def verify blob
         if @peer_key
            sig = nil   
            dig = nil
            blob = nil
            begin
            sig, blob = find K.sig, blob
            dig, blob = find K.digest, blob
            rescue
               log "Message signature was malformed", blob
               return nil
            end
            if @peer_key.sysverify dig, sig
               return blob
            else
               log "Someone has attempted to spoof an email from #{@addr}", blob
               return nil
            end
         else
            # We can't tell...
            text
         end
      end
