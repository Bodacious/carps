def find field, text
   if field.start_with? mark_carp_prefix
      forget, blob = text.split field, 2
      check blob, field
      return ["", blob]
   elsif field.start_with? carp_prefix
      forget, blob = text.split field, 2
      check blob, field
      value, blob = blob.split K.end, 2
      check value, K.end
      return [value, blob] 
   else
      throw StandardError "Invalid keyword"
   end
end
