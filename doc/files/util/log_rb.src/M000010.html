def log reason, *msgs
   $stderr.write "An error was logged:\n"
   $stderr.write reason + "\n"
   msgs.each do |msg|
      $stderr.write msg + "\n"
   end
   $stderr.write "Error raised:\n"
   if $!
      $stderr.write $!.to_s + "\n"
   end
   $stderr.write "End of report.\n"
   puts "\a" 
end
